  <div id="game-container" style="display: none; flex-direction: column; align-items: center;">
    <div id="score">Skor: 0</div>
    <div id="game">
      <div id="character"></div>

  <script>
    const menu = document.getElementById("menu");
    const gameContainer = document.getElementById("game-container");
    const character = document.getElementById("character");
    const obstacle = document.getElementById("obstacle");
    const scoreDisplay = document.getElementById("score");
    const restartDiv = document.getElementById("restart");

    let isJumping = false;
    let jumpHeight = 0;
    let score = 0;
    let gameOver = false;
    let obstaclePosition = 600;
    let animationFrame;

    function startGame() {
      menu.style.display = "none";
      gameContainer.style.display = "flex";
      resetGame();
      moveObstacle();
    }

    function resetGame() {
      isJumping = false;
      jumpHeight = 0;
      score = 0;
      gameOver = false;
      obstaclePosition = 600;
      restartDiv.style.display = "none";
      scoreDisplay.textContent = "Skor: 0";
      character.style.bottom = "50px";
    }

    function jump() {
      if (isJumping || gameOver) return;
      isJumping = true;
      let upInterval = setInterval(() => {
        if (jumpHeight >= 120) {
          clearInterval(upInterval);
          let downInterval = setInterval(() => {
            if (jumpHeight <= 0) {
              clearInterval(downInterval);
              isJumping = false;
              jumpHeight = 0;
            } else {
              jumpHeight -= 5;
              character.style.bottom = 50 + jumpHeight + "px";
            }
          }, 20);
        } else {
          jumpHeight += 5;
          character.style.bottom = 50 + jumpHeight + "px";
        }
      }, 20);
    }

    document.body.onkeydown = (e) => {
      if (e.code === "Space") jump();
    };
    document.body.ontouchstart = () => jump();

    function moveObstacle() {
      if (gameOver) return;
      obstaclePosition -= 5;
      if (obstaclePosition < -60) {
        obstaclePosition = 600;
        score++;
        scoreDisplay.textContent = "Skor: " + score;
      }
      obstacle.style.right = (600 - obstaclePosition) + "px";

      if (
        obstaclePosition < 100 &&
        obstaclePosition > 50 &&
        jumpHeight < 40
      ) {
        gameOver = true;
        restartDiv.style.display = "block";
        scoreDisplay.textContent = "Oyun Bitti! Skor: " + score;
        cancelAnimationFrame(animationFrame);
      } else {
        animationFrame = requestAnimationFrame(moveObstacle);
      }
    }

    function backToMenu() {
      gameContainer.style.display = "none";
      menu.style.display = "flex";
    }
  </script>
